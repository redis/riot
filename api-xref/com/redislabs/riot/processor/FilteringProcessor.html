1   package com.redislabs.riot.processor;
2   
3   import org.springframework.batch.item.ItemProcessor;
4   import org.springframework.expression.Expression;
5   import org.springframework.expression.spel.standard.SpelExpressionParser;
6   import org.springframework.expression.spel.support.StandardEvaluationContext;
7   
8   import java.util.ArrayList;
9   import java.util.Collections;
10  import java.util.List;
11  import java.util.Map;
12  
13  public class FilteringProcessor implements ItemProcessor<Map<String, Object>, Map<String, Object>> {
14  
15      private final StandardEvaluationContext context;
16      private final List<Expression> expressions;
17  
18      public FilteringProcessor(String... filters) {
19          this.context = new StandardEvaluationContext();
20          context.setPropertyAccessors(Collections.singletonList(new MapAccessor()));
21          SpelExpressionParser parser = new SpelExpressionParser();
22          this.expressions = new ArrayList<>();
23          for (String filter : filters) {
24              expressions.add(parser.parseExpression(filter));
25          }
26      }
27  
28      @Override
29      public Map<String, Object> process(Map<String, Object> item) throws Exception {
30          for (Expression expression : expressions) {
31              if (Boolean.FALSE.equals(expression.getValue(context, item))) {
32                  return null;
33              }
34          }
35          return item;
36      }
37  }
