1   package com.redislabs.riot.processor;
2   
3   import org.springframework.batch.item.ExecutionContext;
4   import org.springframework.batch.item.ItemProcessor;
5   import org.springframework.batch.item.ItemStream;
6   import org.springframework.batch.item.ItemStreamException;
7   import org.springframework.batch.item.support.CompositeItemProcessor;
8   
9   import java.util.List;
10  
11  public class CompositeItemStreamItemProcessor<I, O> extends CompositeItemProcessor<I, O> implements ItemStream {
12  
13      private List<? extends ItemProcessor<?, ?>> delegates;
14  
15      @Override
16      public void open(ExecutionContext executionContext) throws ItemStreamException {
17          for (ItemProcessor<?, ?> delegate : delegates) {
18              if (delegate instanceof ItemStream) {
19                  ((ItemStream) delegate).open(executionContext);
20              }
21          }
22      }
23  
24      @Override
25      public void update(ExecutionContext executionContext) throws ItemStreamException {
26          for (ItemProcessor<?, ?> delegate : delegates) {
27              if (delegate instanceof ItemStream) {
28                  ((ItemStream) delegate).update(executionContext);
29              }
30          }
31      }
32  
33      @Override
34      public void close() throws ItemStreamException {
35          for (ItemProcessor<?, ?> delegate : delegates) {
36              if (delegate instanceof ItemStream) {
37                  ((ItemStream) delegate).close();
38              }
39          }
40      }
41  
42      @Override
43      public void setDelegates(List<? extends ItemProcessor<?, ?>> delegates) {
44          super.setDelegates(delegates);
45          this.delegates = delegates;
46      }
47  }
