71      @Override
72      public boolean canRead(EvaluationContext context, @Nullable Object target, String name) throws AccessException {
73          if (target == null) {
74              return false;
75          }
76  
77          Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
78          if (type.isArray() && name.equals("length")) {
79              return true;
80          }
81  
82          PropertyCacheKey cacheKey = new PropertyCacheKey(type, name, target instanceof Class);
83          if (this.readerCache.containsKey(cacheKey)) {
84              return true;
85          }
86          Method method = getMethod(name, type);
87          if (method != null) {
88              TypeDescriptor typeDescriptor = new TypeDescriptor(new MethodParameter(method, -1));
89              this.readerCache.put(cacheKey, new InvokerPair(method, typeDescriptor));
90              this.typeDescriptorCache.put(cacheKey, typeDescriptor);
91              return true;
92          } else {
93              Field field = findField(name, type, target);
94              if (field != null) {
95                  TypeDescriptor typeDescriptor = new TypeDescriptor(field);
96                  this.readerCache.put(cacheKey, new InvokerPair(field, typeDescriptor));
97                  this.typeDescriptorCache.put(cacheKey, typeDescriptor);
98                  return true;
99              }
100         }
101 
102         return false;
103     }

105     private Method getMethod(String name, Class<?> type) {
106         try {
107             return type.getMethod(name);
108         } catch (NoSuchMethodException | SecurityException e) {
109             return null;
110         }
111     }

113     @Override
114     public TypedValue read(EvaluationContext context, @Nullable Object target, String name) throws AccessException {
115         Assert.state(target != null, "Target must not be null");
116         Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
117 
118         if (type.isArray() && name.equals("length")) {
119             if (target instanceof Class) {
120                 throw new AccessException("Cannot access length on array class itself");
121             }
122             return new TypedValue(Array.getLength(target));
123         }
124 
125         PropertyCacheKey cacheKey = new PropertyCacheKey(type, name, target instanceof Class);
126         InvokerPair invoker = this.readerCache.get(cacheKey);
127         this.lastReadInvokerPair = invoker;
128 
129         if (invoker == null || invoker.member instanceof Method) {
130             Method method = (Method) (invoker != null ? invoker.member : null);
131             if (method == null) {
132                 method = getMethod(name, type);
133                 if (method != null) {
134                     // Treat it like a property...
135                     // The readerCache will only contain gettable properties (let's not worry about
136                     // setters for now).
137                     TypeDescriptor typeDescriptor = new TypeDescriptor(new MethodParameter(method, -1));
138                     invoker = new InvokerPair(method, typeDescriptor);
139                     this.lastReadInvokerPair = invoker;
140                     this.readerCache.put(cacheKey, invoker);
141                 }
142             }
143             if (method != null) {
144                 try {
145                     ReflectionUtils.makeAccessible(method);
146                     Object value = method.invoke(target);
147                     return new TypedValue(value, invoker.typeDescriptor.narrow(value));
148                 } catch (Exception ex) {
149                     throw new AccessException("Unable to access property '" + name + "' through getter method", ex);
150                 }
151             }
152         }
153 
154         if (invoker == null || invoker.member instanceof Field) {
155             Field field = (Field) (invoker == null ? null : invoker.member);
156             if (field == null) {
157                 field = findField(name, type, target);
158                 if (field != null) {
159                     invoker = new InvokerPair(field, new TypeDescriptor(field));
160                     this.lastReadInvokerPair = invoker;
161                     this.readerCache.put(cacheKey, invoker);
162                 }
163             }
164             if (field != null) {
165                 try {
166                     ReflectionUtils.makeAccessible(field);
167                     Object value = field.get(target);
168                     return new TypedValue(value, invoker.typeDescriptor.narrow(value));
169                 } catch (Exception ex) {
170                     throw new AccessException("Unable to access field '" + name + "'", ex);
171                 }
172             }
173         }
174 
175         throw new AccessException("Neither getter method nor field found for property '" + name + "'");
176     }

178     @Override
179     public boolean canWrite(EvaluationContext context, @Nullable Object target, String name) throws AccessException {
180         if (!this.allowWrite || target == null) {
181             return false;
182         }
183 
184         Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
185         PropertyCacheKey cacheKey = new PropertyCacheKey(type, name, target instanceof Class);
186         if (this.writerCache.containsKey(cacheKey)) {
187             return true;
188         }
189 
190         Method method = getMethod(name, type);
191         if (method != null) {
192             TypeDescriptor typeDescriptor = new TypeDescriptor(new MethodParameter(method, -1));
193             this.writerCache.put(cacheKey, method);
194             this.typeDescriptorCache.put(cacheKey, typeDescriptor);
195             return true;
196         } else {
197             Field field = findField(name, type, target);
198             if (field != null) {
199                 this.writerCache.put(cacheKey, field);
200                 this.typeDescriptorCache.put(cacheKey, new TypeDescriptor(field));
201                 return true;
202             }
203         }
204 
205         return false;
206     }

208     @Override
209     public void write(EvaluationContext context, @Nullable Object target, String name, @Nullable Object newValue) throws AccessException {
210 
211         if (!this.allowWrite) {
212             throw new AccessException("PropertyAccessor for property '" + name + "' on target [" + target + "] does not allow write operations");
213         }
214 
215         Assert.state(target != null, "Target must not be null");
216         Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
217 
218         Object possiblyConvertedNewValue = newValue;
219         TypeDescriptor typeDescriptor = getTypeDescriptor(context, target, name);
220         if (typeDescriptor != null) {
221             try {
222                 possiblyConvertedNewValue = context.getTypeConverter().convertValue(newValue, TypeDescriptor.forObject(newValue), typeDescriptor);
223             } catch (EvaluationException evaluationException) {
224                 throw new AccessException("Type conversion failure", evaluationException);
225             }
226         }
227 
228         PropertyCacheKey cacheKey = new PropertyCacheKey(type, name, target instanceof Class);
229         Member cachedMember = this.writerCache.get(cacheKey);
230 
231         if (cachedMember == null || cachedMember instanceof Method) {
232             Method method = (Method) cachedMember;
233             if (method != null) {
234                 try {
235                     ReflectionUtils.makeAccessible(method);
236                     method.invoke(target, possiblyConvertedNewValue);
237                     return;
238                 } catch (Exception ex) {
239                     throw new AccessException("Unable to access property '" + name + "' through setter method", ex);
240                 }
241             }
242         }
243 
244         if (cachedMember == null || cachedMember instanceof Field) {
245             Field field = (Field) cachedMember;
246             if (field == null) {
247                 field = findField(name, type, target);
248                 if (field != null) {
249                     cachedMember = field;
250                     this.writerCache.put(cacheKey, cachedMember);
251                 }
252             }
253             if (field != null) {
254                 try {
255                     ReflectionUtils.makeAccessible(field);
256                     field.set(target, possiblyConvertedNewValue);
257                     return;
258                 } catch (Exception ex) {
259                     throw new AccessException("Unable to access field '" + name + "'", ex);
260                 }
261             }
262         }
263 
264         throw new AccessException("Neither setter method nor field found for property '" + name + "'");
265     }

279     @Nullable
280     private TypeDescriptor getTypeDescriptor(EvaluationContext context, Object target, String name) {
281         Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
282 
283         if (type.isArray() && name.equals("length")) {
284             return TypeDescriptor.valueOf(Integer.TYPE);
285         }
286         PropertyCacheKey cacheKey = new PropertyCacheKey(type, name, target instanceof Class);
287         TypeDescriptor typeDescriptor = this.typeDescriptorCache.get(cacheKey);
288         if (typeDescriptor == null) {
289             // Attempt to populate the cache entry
290             try {
291                 if (canRead(context, target, name) || canWrite(context, target, name)) {
292                     typeDescriptor = this.typeDescriptorCache.get(cacheKey);
293                 }
294             } catch (AccessException ex) {
295                 // Continue with null type descriptor
296             }
297         }
298         return typeDescriptor;
299     }

316     @Nullable
317     private Field findField(String name, Class<?> clazz, Object target) {
318         Field field = findField(name, clazz, target instanceof Class);
319         if (field == null && target instanceof Class) {
320             field = findField(name, target.getClass(), false);
321         }
322         return field;
323     }

325     /**
326      * Find a field of a certain name on a specified class.
327      */
328     @Nullable
329     protected Field findField(String name, Class<?> clazz, boolean mustBeStatic) {
330         Field[] fields = clazz.getFields();
331         for (Field field : fields) {
332             if (field.getName().equals(name) && (!mustBeStatic || Modifier.isStatic(field.getModifiers()))) {
333                 return field;
334             }
335         }
336         // We'll search superclasses and implemented interfaces explicitly,
337         // although it shouldn't be necessary - however, see SPR-10125.
338         if (clazz.getSuperclass() != null) {
339             Field field = findField(name, clazz.getSuperclass(), mustBeStatic);
340             if (field != null) {
341                 return field;
342             }
343         }
344         for (Class<?> implementedInterface : clazz.getInterfaces()) {
345             Field field = findField(name, implementedInterface, mustBeStatic);
346             if (field != null) {
347                 return field;
348             }
349         }
350         return null;
351     }

353     /**
354      * Attempt to create an optimized property accessor tailored for a property of a
355      * particular name on a particular class. The general ReflectivePropertyAccessor
356      * will always work but is not optimal due to the need to lookup which
357      * reflective member (method/field) to use each time read() is called. This
358      * method will just return the ReflectivePropertyAccessor instance if it is
359      * unable to build a more optimal accessor.
360      * <p>
361      * Note: An optimal accessor is currently only usable for read attempts. Do not
362      * call this method if you need a read-write accessor.
363      *
364      * @see OptimalPropertyAccessor
365      */
366     public PropertyAccessor createOptimalAccessor(@Nullable Object target, String name) {
367         // Don't be clever for arrays or a null target...
368         if (target == null) {
369             return this;
370         }
371         Class<?> clazz = (target instanceof Class ? (Class<?>) target : target.getClass());
372         if (clazz.isArray()) {
373             return this;
374         }
375 
376         PropertyCacheKey cacheKey = new PropertyCacheKey(clazz, name, target instanceof Class);
377         InvokerPair invocationTarget = this.readerCache.get(cacheKey);
378 
379         if (invocationTarget == null || invocationTarget.member instanceof Method) {
380             Method method = (Method) (invocationTarget != null ? invocationTarget.member : null);
381             if (method == null) {
382                 method = getMethod(name, clazz);
383                 if (method != null) {
384                     invocationTarget = new InvokerPair(method, new TypeDescriptor(new MethodParameter(method, -1)));
385                     ReflectionUtils.makeAccessible(method);
386                     this.readerCache.put(cacheKey, invocationTarget);
387                 }
388             }
389             if (method != null) {
390                 return new OptimalPropertyAccessor(invocationTarget);
391             }
392         }
393 
394         if (invocationTarget == null || invocationTarget.member instanceof Field) {
395             Field field = (invocationTarget != null ? (Field) invocationTarget.member : null);
396             if (field == null) {
397                 field = findField(name, clazz, target instanceof Class);
398                 if (field != null) {
399                     invocationTarget = new InvokerPair(field, new TypeDescriptor(field));
400                     ReflectionUtils.makeAccessible(field);
401                     this.readerCache.put(cacheKey, invocationTarget);
402                 }
403             }
404             if (field != null) {
405                 return new OptimalPropertyAccessor(invocationTarget);
406             }
407         }
408 
409         return this;
410     }

414      * value and the type descriptor for the value returned by the reflective call.

416     private static class InvokerPair {
417 
418         final Member member;
419 
420         final TypeDescriptor typeDescriptor;
421 
422         public InvokerPair(Member member, TypeDescriptor typeDescriptor) {
423             this.member = member;
424             this.typeDescriptor = typeDescriptor;

428     private static final class PropertyCacheKey implements Comparable<PropertyCacheKey> {
429 
430         private final Class<?> clazz;
431         private final String property;
432         private final boolean targetIsClass;
433 
434         public PropertyCacheKey(Class<?> clazz, String name, boolean targetIsClass) {
435             this.clazz = clazz;
436             this.property = name;
437             this.targetIsClass = targetIsClass;
438         }
439 
440         @Override
441         public boolean equals(Object other) {
442             if (this == other) {
443                 return true;
444             }
445             if (!(other instanceof PropertyCacheKey)) {
446                 return false;
447             }
448             PropertyCacheKey otherKey = (PropertyCacheKey) other;
449             return (this.clazz == otherKey.clazz && this.property.equals(otherKey.property) && this.targetIsClass == otherKey.targetIsClass);
450         }
451 
452         @Override
453         public int hashCode() {
454             return (this.clazz.hashCode() * 29 + this.property.hashCode());
455         }
456 
457         @Override
458         public String toString() {
459             return "CacheKey [clazz=" + this.clazz.getName() + ", property=" + this.property + ", " + this.property + ", targetIsClass=" + this.targetIsClass + "]";
460         }
461 
462         @Override
463         public int compareTo(PropertyCacheKey other) {
464             int result = this.clazz.getName().compareTo(other.clazz.getName());
465             if (result == 0) {
466                 result = this.property.compareTo(other.property);
467             }
468             return result;
469         }
470     }

472     /**
473      * An optimized form of a PropertyAccessor that will use reflection but only
474      * knows how to access a particular property on a particular class. This is
475      * unlike the general ReflectivePropertyResolver which manages a cache of
476      * methods/fields that may be invoked to access different properties on
477      * different classes. This optimal accessor exists because looking up the
478      * appropriate reflective object by class/name on each read is not cheap.
479      */
480     public static class OptimalPropertyAccessor implements CompilablePropertyAccessor {
481 
482         /**
483          * The member being accessed.
484          */
485         public final Member member;
486 
487         private final TypeDescriptor typeDescriptor;
488 
489         OptimalPropertyAccessor(InvokerPair target) {
490             this.member = target.member;
491             this.typeDescriptor = target.typeDescriptor;
492         }
493 
494         @Override
495         @Nullable
496         public Class<?>[] getSpecificTargetClasses() {
497             throw new UnsupportedOperationException("Should not be called on an OptimalPropertyAccessor");
498         }
499 
500         @Override
501         public boolean canRead(EvaluationContext context, @Nullable Object target, String name) {
502             if (target == null) {
503                 return false;
504             }
505             Class<?> type = (target instanceof Class ? (Class<?>) target : target.getClass());
506             if (type.isArray()) {
507                 return false;
508             }
509 
510             if (this.member instanceof Method) {
511                 Method method = (Method) this.member;
512                 return name.equals(method.getName());
513             } else {
514                 Field field = (Field) this.member;
515                 return field.getName().equals(name);
516             }
517         }
518 
519         @Override
520         public TypedValue read(EvaluationContext context, @Nullable Object target, String name) throws AccessException {
521             if (this.member instanceof Method) {
522                 Method method = (Method) this.member;
523                 try {
524                     ReflectionUtils.makeAccessible(method);
525                     Object value = method.invoke(target);
526                     return new TypedValue(value, this.typeDescriptor.narrow(value));
527                 } catch (Exception ex) {
528                     throw new AccessException("Unable to access property '" + name + "' through getter method", ex);
529                 }
530             } else {
531                 Field field = (Field) this.member;
532                 try {
533                     ReflectionUtils.makeAccessible(field);
534                     Object value = field.get(target);
535                     return new TypedValue(value, this.typeDescriptor.narrow(value));
536                 } catch (Exception ex) {
537                     throw new AccessException("Unable to access field '" + name + "'", ex);
538                 }
539             }
540         }
541 
542         @Override
543         public boolean canWrite(EvaluationContext context, @Nullable Object target, String name) {
544             throw new UnsupportedOperationException("Should not be called on an OptimalPropertyAccessor");
545         }
546 
547         @Override
548         public void write(EvaluationContext context, @Nullable Object target, String name, @Nullable Object newValue) {
549             throw new UnsupportedOperationException("Should not be called on an OptimalPropertyAccessor");
550         }
551 
552         @Override
553         public boolean isCompilable() {
554             return (Modifier.isPublic(this.member.getModifiers()) && Modifier.isPublic(this.member.getDeclaringClass().getModifiers()));
555         }
556 
557         @Override
558         public Class<?> getPropertyType() {
559             if (this.member instanceof Method) {
560                 return ((Method) this.member).getReturnType();
561             } else {
562                 return ((Field) this.member).getType();
563             }
564         }
565 
566         @Override
567         public void generateCode(String propertyName, MethodVisitor mv, CodeFlow cf) {
568             boolean isStatic = Modifier.isStatic(this.member.getModifiers());
569             String descriptor = cf.lastDescriptor();
570             String classDesc = this.member.getDeclaringClass().getName().replace('.', '/');
571 
572             if (!isStatic) {
573                 if (descriptor == null) {
574                     cf.loadTarget(mv);
575                 }
576                 if (descriptor == null || !classDesc.equals(descriptor.substring(1))) {
577                     mv.visitTypeInsn(CHECKCAST, classDesc);
578                 }
579             } else {
580                 if (descriptor != null) {
581                     // A static field/method call will not consume what is on the stack,
582                     // it needs to be popped off.
583                     mv.visitInsn(POP);
584                 }
585             }
586 
587             if (this.member instanceof Method) {
588                 mv.visitMethodInsn((isStatic ? INVOKESTATIC : INVOKEVIRTUAL), classDesc, this.member.getName(), CodeFlow.createSignatureDescriptor((Method) this.member), false);
589             } else {
590                 mv.visitFieldInsn((isStatic ? GETSTATIC : GETFIELD), classDesc, this.member.getName(), CodeFlow.toJvmDescriptor(((Field) this.member).getType()));
591             }
592         }
593     }
