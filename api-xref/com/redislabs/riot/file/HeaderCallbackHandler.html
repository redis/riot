1   package com.redislabs.riot.file;
2   
3   import lombok.extern.slf4j.Slf4j;
4   import org.springframework.batch.item.file.LineCallbackHandler;
5   import org.springframework.batch.item.file.transform.AbstractLineTokenizer;
6   import org.springframework.batch.item.file.transform.FieldSet;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  @Slf4j
12  public class HeaderCallbackHandler implements LineCallbackHandler {
13  
14      private final AbstractLineTokenizer tokenizer;
15  
16      public HeaderCallbackHandler(AbstractLineTokenizer tokenizer) {
17          this.tokenizer = tokenizer;
18      }
19  
20      @Override
21      public void handleLine(String line) {
22          log.debug("Found header {}", line);
23          FieldSet fieldSet = tokenizer.tokenize(line);
24          List<String> fields = new ArrayList<>();
25          for (int index = 0; index < fieldSet.getFieldCount(); index++) {
26              fields.add(fieldSet.readString(index));
27          }
28          tokenizer.setNames(fields.toArray(new String[0]));
29      }
30  }
