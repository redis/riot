24  public class FileUtils {
25  
26      public final static String GS_URI_PREFIX = "gs://";
27      public final static String S3_URI_PREFIX = "s3://";
28  
29      private final static Pattern EXTENSION_PATTERN = Pattern.compile("(?i)\\.(?<extension>\\w+)(?<gz>\\.gz)?$");
30      public static final String EXTENSION_CSV = "csv";
31      public static final String EXTENSION_TSV = "tsv";
32      public static final String EXTENSION_PSV = "psv";
33      public static final String EXTENSION_FW = "fw";
34      public static final String EXTENSION_JSON = "json";
35      public static final String EXTENSION_XML = "xml";
36  
37      public static boolean isGzip(String file) {
38          return extensionGroup(file, "gz") != null;
39      }
40  
41      public static String extension(String file) {
42          return extensionGroup(file, "extension");
43      }
44  
45      public static String extensionGroup(String file, String group) {
46          Matcher matcher = EXTENSION_PATTERN.matcher(file);
47          if (matcher.find()) {
48              return matcher.group(group);
49          }
50          return null;
51      }
52  
53      public static boolean isFile(String file) {
54          return !(isGcs(file) || isS3(file) || ResourceUtils.isUrl(file) || isConsole(file));
55      }
56  
57      public static boolean isConsole(String file) {
58          return "-".equals(file);
59      }
60  
61      public static boolean isS3(String file) {
62          return file.startsWith(S3_URI_PREFIX);
63      }
64  
65      public static boolean isGcs(String file) {
66          return file.startsWith(GS_URI_PREFIX);
67      }
68  
69      public static <T> JsonItemReader<T> jsonReader(Resource resource, Class<T> clazz) {
70          JsonItemReaderBuilder<T> jsonReaderBuilder = new JsonItemReaderBuilder<>();
71          jsonReaderBuilder.name(resource.getFilename() + "-json-file-reader");
72          jsonReaderBuilder.resource(resource);
73          JacksonJsonObjectReader<T> jsonObjectReader = new JacksonJsonObjectReader<>(clazz);
74          jsonObjectReader.setMapper(new ObjectMapper());
75          jsonReaderBuilder.jsonObjectReader(jsonObjectReader);
76          return jsonReaderBuilder.build();
77      }
78  
79      public static <T> XmlItemReader<T> xmlReader(Resource resource, Class<T> clazz) {
80          XmlItemReaderBuilder<T> xmlReaderBuilder = new XmlItemReaderBuilder<>();
81          xmlReaderBuilder.name(resource.getFilename() + "-xml-file-reader");
82          xmlReaderBuilder.resource(resource);
83          XmlObjectReader<T> xmlObjectReader = new XmlObjectReader<>(clazz);
84          xmlObjectReader.setMapper(new XmlMapper());
85          xmlReaderBuilder.xmlObjectReader(xmlObjectReader);
86          return xmlReaderBuilder.build();
87      }
88  
89      public static List<String> expand(List<String> files) throws IOException {
90          if (files == null) {
91              return null;
92          }
93          List<String> expandedFiles = new ArrayList<>();
94          for (String file : files) {
95              if (isFile(file)) {
96                  Path path = Paths.get(file);
97                  if (Files.exists(path)) {
98                      expandedFiles.add(file);
99                  } else {
100                     // Path might be glob pattern
101                     Path parent = path.getParent();
102                     if (parent != null && Files.exists(parent)) {
103                         try (DirectoryStream<Path> stream = Files.newDirectoryStream(parent, path.getFileName().toString())) {
104                             stream.forEach(p -> expandedFiles.add(p.toString()));
105                         }
106                     }
107                 }
108             } else {
109                 expandedFiles.add(file);
110             }
111         }
112         return expandedFiles;
113     }
