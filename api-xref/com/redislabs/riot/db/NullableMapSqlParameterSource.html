3   import org.springframework.jdbc.core.namedparam.MapSqlParameterSource;
4   import org.springframework.lang.Nullable;
5   
6   import java.util.Map;
7   
8   public class NullableMapSqlParameterSource extends MapSqlParameterSource {
9   
10      public NullableMapSqlParameterSource() {
11          super();
12      }
13  
14      public NullableMapSqlParameterSource(String paramName, @Nullable Object value) {
15          super(paramName, value);
16      }
17  
18      public NullableMapSqlParameterSource(@Nullable Map<String, ?> values) {
19          super(values);
20      }
21  
22      @Override
23      @Nullable
24      public Object getValue(String paramName) {
25          if (!hasValue(paramName)) {
26              return null;
27          }
28          return super.getValue(paramName);
29      }
30  
31  
32  }
