1   package com.redislabs.riot.db;
2   
3   import org.springframework.boot.autoconfigure.jdbc.DataSourceProperties;
4   import picocli.CommandLine.Option;
5   
6   import javax.sql.DataSource;
7   
8   public class DataSourceOptions {
9   
10      @Option(names = "--driver", description = "Fully qualified name of the JDBC driver", paramLabel = "<class>")
11      private String driver;
12      @Option(names = "--url", required = true, description = "JDBC URL to connect to the database", paramLabel = "<string>")
13      private String url;
14      @Option(names = "--username", description = "Login username of the database", paramLabel = "<string>")
15      private String username;
16      @Option(names = "--password", arity = "0..1", interactive = true, description = "Login password of the database", paramLabel = "<pwd>")
17      private String password;
18  
19      public String getDriver() {
20          return driver;
21      }
22  
23      public void setDriver(String driver) {
24          this.driver = driver;
25      }
26  
27      public String getUrl() {
28          return url;
29      }
30  
31      public void setUrl(String url) {
32          this.url = url;
33      }
34  
35      public String getUsername() {
36          return username;
37      }
38  
39      public void setUsername(String username) {
40          this.username = username;
41      }
42  
43      public String getPassword() {
44          return password;
45      }
46  
47      public void setPassword(String password) {
48          this.password = password;
49      }
50  
51      public DataSource dataSource() {
52          DataSourceProperties properties = new DataSourceProperties();
53          properties.setUrl(url);
54          properties.setDriverClassName(driver);
55          properties.setUsername(username);
56          properties.setPassword(password);
57          return properties.initializeDataSourceBuilder().build();
58      }
59  
60  }
