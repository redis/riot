1   package com.redislabs.riot;
2   
3   import lombok.Data;
4   import picocli.CommandLine;
5   
6   import java.time.Duration;
7   
8   @Data
9   public class FlushingTransferOptions {
10  
11      @CommandLine.Option(names = "--flush-interval", description = "Max duration between flushes (default: ${DEFAULT-VALUE})", paramLabel = "<ms>")
12      private long flushInterval = 50;
13      @CommandLine.Option(names = "--idle-timeout", description = "Min duration of inactivity to consider transfer complete", paramLabel = "<ms>")
14      private Long idleTimeout;
15  
16      public Duration getFlushIntervalDuration() {
17          return Duration.ofMillis(flushInterval);
18      }
19  
20      public void setIdleTimeout(Duration idleTimeoutDuration) {
21          if (idleTimeoutDuration == null) {
22              this.idleTimeout = null;
23          } else {
24              this.idleTimeout = idleTimeoutDuration.toMillis();
25          }
26      }
27  
28      public Duration getIdleTimeoutDuration() {
29          if (idleTimeout == null) {
30              return null;
31          }
32          return Duration.ofMillis(idleTimeout);
33      }
34  
35  }
