1   package com.redislabs.riot;
2   
3   import com.redislabs.mesclun.RedisModulesClient;
4   import io.lettuce.core.cluster.RedisClusterClient;
5   import lombok.extern.slf4j.Slf4j;
6   import org.springframework.batch.core.configuration.annotation.StepBuilderFactory;
7   import org.springframework.batch.core.step.builder.AbstractTaskletStepBuilder;
8   import org.springframework.batch.core.step.builder.SimpleStepBuilder;
9   import org.springframework.batch.core.step.builder.StepBuilder;
10  import org.springframework.batch.item.ItemProcessor;
11  import org.springframework.batch.item.ItemWriter;
12  import org.springframework.batch.item.redis.DataStructureItemReader;
13  import org.springframework.batch.item.redis.support.DataStructure;
14  import picocli.CommandLine;
15  
16  @Slf4j
17  public abstract class AbstractExportCommand<O> extends AbstractTransferCommand {
18  
19      @CommandLine.ArgGroup(exclusive = false, heading = "Redis reader options%n")
20      private RedisReaderOptionsedisReaderOptions">RedisReaderOptions options = new RedisReaderOptions();
21  
22      protected AbstractTaskletStepBuilder<SimpleStepBuilder<DataStructure, O>> step(StepBuilderFactory stepBuilderFactory, ItemProcessor<DataStructure, O> processor, ItemWriter<O> writer) throws Exception {
23          String name = name(getRedisOptions().uris().get(0));
24          StepBuilder stepBuilder = stepBuilderFactory.get(name + "-export-step");
25          RiotStepBuilder<DataStructure, O> step = riotStep(stepBuilder, "Exporting from " + name);
26          return step.reader(reader()).processor(processor).writer(writer).build();
27      }
28  
29      protected final DataStructureItemReader reader() {
30          RedisOptions redisOptions = getRedisOptions();
31          if (redisOptions.isCluster()) {
32              RedisClusterClient client = redisOptions.redisClusterClient();
33              return options.configure(DataStructureItemReader.client(client)).build();
34          }
35          RedisModulesClient client = redisOptions.redisClient();
36          return options.configure(DataStructureItemReader.client(client)).build();
37      }
38  
39      @Override
40      protected <I, O> RiotStepBuilder<I, O> riotStep(StepBuilder stepBuilder, String taskName) {
41          RiotStepBuilder<I, O> riotStepBuilder = super.riotStep(stepBuilder, taskName);
42          riotStepBuilder.initialMax(options.initialMaxSupplier(getRedisOptions()));
43          return riotStepBuilder;
44      }
45  
46  }
