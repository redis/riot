1   package com.redislabs.riot.convert;
2   
3   import org.springframework.core.convert.converter.Converter;
4   import org.springframework.util.NumberUtils;
5   
6   public class ObjectToNumberConverter<T extends Number> implements Converter<Object, T> {
7   
8       private final Class<T> targetType;
9   
10      public ObjectToNumberConverter(Class<T> targetType) {
11          this.targetType = targetType;
12      }
13  
14      @Override
15      public T convert(Object source) {
16          if (source == null) {
17              return null;
18          }
19          if (source instanceof Number) {
20              return NumberUtils.convertNumberToTargetClass((Number) source, targetType);
21          }
22          if (source instanceof String) {
23              String string = (String) source;
24              if (string.isEmpty()) {
25                  return null;
26              }
27              return NumberUtils.parseNumber(string, targetType);
28          }
29          return null;
30      }
31  
32  }
