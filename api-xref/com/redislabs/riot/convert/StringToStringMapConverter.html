1   package com.redislabs.riot.convert;
2   
3   import lombok.Data;
4   import org.springframework.core.convert.converter.Converter;
5   
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   @Data
10  public class StringToStringMapConverter implements Converter<String, Map<String, String>> {
11  
12      public static final Converter<String, String> DEFAULT_KEY_EXTRACTOR = s -> "value";
13  
14      private Converter<String, String> keyExtractor = DEFAULT_KEY_EXTRACTOR;
15  
16      @Override
17      public Map<String, String> convert(String source) {
18          Map<String, String> result = new HashMap<>();
19          result.put(keyExtractor.convert(source), source);
20          return result;
21      }
22  
23  }
