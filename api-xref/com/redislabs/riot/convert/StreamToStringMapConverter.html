1   package com.redislabs.riot.convert;
2   
3   import io.lettuce.core.StreamMessage;
4   import lombok.Builder;
5   import org.springframework.core.convert.converter.Converter;
6   
7   import java.util.HashMap;
8   import java.util.List;
9   import java.util.Map;
10  
11  @Builder
12  public class StreamToStringMapConverter implements Converter<List<StreamMessage<String, String>>, Map<String, String>> {
13  
14  	public static final String DEFAULT_KEY_FORMAT = "%s.%s";
15  
16  	@Builder.Default
17  	private final String keyFormat = DEFAULT_KEY_FORMAT;
18  
19  	@Override
20  	public Map<String, String> convert(List<StreamMessage<String, String>> source) {
21  		Map<String, String> result = new HashMap<>();
22  		for (StreamMessage<String, String> message : source) {
23  			for (Map.Entry<String, String> entry : message.getBody().entrySet()) {
24  				result.put(String.format(keyFormat, message.getId(), entry.getKey()), entry.getValue());
25  			}
26  		}
27  		return result;
28  	}
29  }
